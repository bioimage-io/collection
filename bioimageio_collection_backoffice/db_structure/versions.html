245    def get_updated(self, update: Versions) -> Versions:
246        if update.doi is None:
247            concept_doi = self.doi
248        elif self.doi is None:
249            concept_doi = update.doi
250        elif self.doi != update.doi:
251            raise ValueError("May not overwrite doi")
252        else:
253            concept_doi = self.doi
254
255        return Versions(

257            staged={**self.staged, **update.staged},
