328    def unpack(self, package_url: str):
329        previous_version = self.concept.get_latest_published_version()
330        if previous_version is None:
331            previous_rdf = None
332        else:
333            previous_rdf_data = self.client.load_file(previous_version.rdf_path)
334            if previous_rdf_data is None:
335                self.set_error_status("Failed to load previous published version's RDF")
336                sys.exit(1)
337
338            previous_rdf: Optional[Dict[Any, Any]] = yaml.load(
339                io.BytesIO(previous_rdf_data)
340            )
341            assert isinstance(previous_rdf, dict)
342
343        # ensure we have a chat.json
344        self.extend_chat(Chat())
345
346        # ensure we have a log.json
347        self.extend_log(Log())
348
349        # set first status (this also write versions.json)
350        self._set_status(
351            UnpackingStatus(description=f"unzipping {package_url} to {self.folder}")
352        )
353
354        # Download the model zip file
355        try:
356            remotezip = urllib.request.urlopen(package_url)
357        except Exception as e:
358            self.set_error_status(f"failed to open {package_url}: {e}")
359            sys.exit(1)
360
361        zipinmemory = io.BytesIO(remotezip.read())
362
363        # Unzip the zip file
364        zipobj = zipfile.ZipFile(zipinmemory)
365
366        file_names = set(zipobj.namelist())
367        bioimageio_yaml_file_name = identify_bioimageio_yaml_file_name(file_names)
368
369        rdf: Union[Any, Dict[Any, Any]] = yaml.load(
370            zipobj.open(bioimageio_yaml_file_name).read().decode()
371        )
372        if not isinstance(rdf, dict):
373            self.set_error_status(
374                f"Expected {bioimageio_yaml_file_name} to hold a dictionary"
375            )
376            sys.exit(1)
377
378        if (rdf_id := rdf.get("id")) is None:
379            rdf["id"] = self.id
380        elif rdf_id != self.id:
381            self.set_error_status(
382                f"Expected package for {self.id}, "
383                + f"but got packaged {rdf_id} ({package_url})"
384            )
385            sys.exit(1)
386
387        if "name" not in rdf:
388            self.set_error_status(f"Missing 'name' in {package_url}")
389            sys.exit(1)
390
391        collection_data = self.client.load_file("collection.json")
392        assert collection_data is not None
393        collection = json.loads(collection_data)
394        for e in collection["collection"]:
395            if e["name"] == rdf["name"]:
396                if e["id"] != rdf["id"]:
397                    self.extend_log(
398                        Log(
399                            collection=[
400                                CollectionLog(
401                                    log=CollectionLogEntry(
402                                        message=f"Another resource with name='{rdf['name']}' already exists ({e['id']})"
403                                    )
404                                )
405                            ]
406                        )
407                    )
408                break
409
410        # set matching id_emoji
411        id_parts = IdParts.load()
412        rdf["id_emoji"] = id_parts.get_icon(self.id)
413        if rdf["id_emoji"] is None:
414            self.extend_log(
415                Log(
416                    collection=[
417                        CollectionLog(
418                            log=CollectionLogEntry(
419                                message="Failed to get icon for {self.id}"
420                            )
421                        )
422                    ]
423                )
424            )
425
426        # overwrite version information
427        rdf["version_number"] = self.number
428
429        if "id" not in rdf:
430            raise ValueError("Missing `id` field")
431
432        if not str(rdf["id"]):
433            raise ValueError(f"Invalid `id`: {rdf['id']}")
434
435        if (
436            "uploader" not in rdf
437            or not isinstance(rdf["uploader"], dict)
438            or "email" not in rdf["uploader"]
439        ):
440            self.set_error_status("RDF is missing `uploader.email` field.")
441            sys.exit(1)
442        elif not isinstance(rdf["uploader"]["email"], str):
443            self.set_error_status("RDF has invalid `uploader.email` field.")
444            sys.exit(1)
445
446        uploader = rdf["uploader"]["email"]
447        if previous_rdf is not None:
448            prev_authors: List[Dict[str, str]] = previous_rdf["authors"]
449            assert isinstance(prev_authors, list)
450            prev_maintainers: List[Dict[str, str]] = (
451                previous_rdf.get("maintainers", []) + prev_authors
452            )
453            maintainer_emails = [a["email"] for a in prev_maintainers if "email" in a]
454            if (
455                uploader != previous_rdf.get("uploader", {}).get("email", BOT_EMAIL)
456                and uploader not in maintainer_emails
457                and uploader not in [r.email for r in get_reviewers().values()]
458            ):
459                self.set_error_status(
460                    f"uploader '{uploader}' is not a maintainer of {self.id} nor a registered bioimageio reviewer."
461                )
462                sys.exit()
463
464        def upload(file_name: str, file_data: bytes):
465            path = f"{self.folder}files/{file_name}"
466            self.client.put(path, io.BytesIO(file_data), length=len(file_data))
467
468        thumbnails = create_thumbnails(rdf, zipobj)
469        config = rdf.setdefault("config", {})
470        if isinstance(config, dict):
471            bioimageio_config: Any = config.setdefault("bioimageio", {})
472            if isinstance(bioimageio_config, dict):
473                thumbnail_config: Any = bioimageio_config.setdefault("thumbnails", {})
474                if isinstance(thumbnail_config, dict):
475                    for oname, (tname, tdata) in thumbnails.items():
476                        upload(tname, tdata)
477                        thumbnail_config[oname] = tname
478
479        rdf_io = io.BytesIO()
480        yaml.dump(rdf, rdf_io)
481        rdf_data = rdf_io.getvalue()
482        upload("bioimageio.yaml", rdf_data)
483        upload("rdf.yaml", rdf_data)
484
485        file_names.remove(bioimageio_yaml_file_name)
486        for other in {fn for fn in file_names if is_valid_bioimageio_yaml_name(fn)}:
487            logger.warning("ignoring alternative bioimageio.yaml source '{other}'")
488            file_names.remove(other)
489
490        for file_name in file_names:
491            file_data = zipobj.open(file_name).read()
492            upload(file_name, file_data)

494        self._set_status(UnpackedStatus())
495        self.supersede_previously_staged_versions()

569            next_publish_nr = PublishNumber(max(self.concept.versions.published) + 1)

582        sem_ver = rdf.get("version")

584            sem_ver = str(sem_ver)

606            staged={
607                self.number: StagedVersionInfo(
608                    sem_ver=self.concept.versions.staged[self.number].sem_ver,
609                    timestamp=self.concept.versions.staged[self.number].timestamp,
610                    status=PublishedStagedStatus(publish_number=next_publish_nr),
611                )
612            },
613            published={
614                next_publish_nr: PublishedVersionInfo(
615                    sem_ver=sem_ver,
616                    status=PublishedStatus(stage_number=self.number),
617                )
618            },
619        )
620        self.concept.extend_versions(verions_update)

622        # TODO: clean up staged files?
623        # remove all uploaded files from this staged version
624        # self.client.rm_dir(f"{self.folder}/files/")
