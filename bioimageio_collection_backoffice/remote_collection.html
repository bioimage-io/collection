 187def log(
 188    func: Callable[Concatenate[R, P], T],
 189) -> Callable[Concatenate[R, P], T]:
 190    """method decorator to indicate that a method execution should be logged"""
 191
 192    @wraps(func)
 193    def wrapper(self: R, *args: P.args, **kwargs: P.kwargs):
 194        self.log_message(f"starting: '{func.__name__}'")
 195        try:
 196            ret = func(self, *args, **kwargs)
 197        except Exception as e:
 198            self.log_error(e)
 199            raise
 200        else:
 201            return ret
 202
 203    return wrapper

 208) -> Callable[Concatenate[R, str, P], T]:
 209    """method decorator to indicate that a method may only be called by a bioimage.io reviewer"""

 221def lock_concept(
 222    func: Callable[Concatenate[R, P], T],
 223) -> Callable[Concatenate[R, P], T]:
 224    """method decorator to indicate that a method may only be called by a bioimage.io reviewer"""
 225
 226    @wraps(func)
 227    def wrapper(self: R, *args: P.args, **kwargs: P.kwargs):
 228        concept_id = self.concept_id
 229        assert not concept_id.endswith("/"), concept_id
 230        lock_path = f"{concept_id}/concept-lock"
 231        if list(self.client.ls(lock_path)):
 232            raise ValueError(f"{concept_id} is currently locked")
 233
 234        self.client.put(lock_path, io.BytesIO(b" "), length=1)
 235        try:
 236            return func(self, *args, **kwargs)
 237        finally:
 238            self.client.rm(lock_path)
 239
 240    return wrapper
